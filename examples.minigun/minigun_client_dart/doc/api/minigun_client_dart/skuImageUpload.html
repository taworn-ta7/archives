Future<MixResult> skuImageUpload(
  Client client, {
  required String code,
  required http.MultipartFile file,
}) async {
  final req = client.prepareMultipart(
    'POST',
    client.getUri('sku/image/$code'),
  );
  req.files.add(file);
  return await client.callMultipart(req);
}
