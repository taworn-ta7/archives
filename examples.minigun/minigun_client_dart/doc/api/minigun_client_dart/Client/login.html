Future<MixResult> login({
  required String username,
  required String password,
}) async {
  final json = {
    'login': {
      'username': username,
      'password': password,
    },
  };
  final result = await call(
    MethodType.put,
    getUri('authen/login'),
    body: jsonEncode(json),
  );
  if (result.isOk() && result.rest!.ok) {
    final rest = result.rest!;
    _token = rest.json['user']['token'];
  }
  return result;
}
